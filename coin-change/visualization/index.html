Step 6 — Complete Solution

O(total × coins) time  ·  O(total) space

def coin_change(coins, total):
  dp = [total + 1] * (total + 1)
  dp[0] = 0
  for amount in range(1, total + 1):
    for coin in coins:
      if coin <= amount:
        dp[amount] = min(
          dp[amount],
          dp[amount − coin] + 1
        )
  return dp[total] if dp[total] <= total else −1
